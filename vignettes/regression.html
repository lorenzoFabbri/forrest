Forest plots for regression results

Forest plots are not limited to meta-analyses. Any table of point estimates with confidence intervals can be displayed as a forest plot — results from regression models, dose-response analyses, multi-outcome studies, and more.

This vignette shows how to go from a fitted R model to a forest plot in a few lines of code using broom::tidy() to extract parameters. broom is not a hard dependency of forrest; it is used here solely as a convenient extraction tool. You can pass any data frame with the right columns directly to forrest().

Simulated data

set.seed(42)
n <- 500

cohort <- data.frame(
  sbp      = 120 + rnorm(n, sd = 15),
  age      = runif(n, 30, 75),
  female   = rbinom(n, 1, 0.5),
  bmi      = rnorm(n, 26, 4),
  smoker   = rbinom(n, 1, 0.2),
  activity = rbinom(n, 1, 0.4)
)
# Introduce real signal
cohort$sbp <- cohort$sbp +
  0.35 * cohort$age -
  5.0  * cohort$female +
  0.4  * cohort$bmi -
  2.0  * cohort$smoker +
  2.0  * cohort$activity +
  rnorm(n, sd = 6)

# Binary outcome (hypertension: SBP ≥ 140)
cohort$htn <- as.integer(cohort$sbp >= 140)

Multiple predictors from one model

The simplest workflow: fit a model, call broom::tidy(conf.int = TRUE), and pass the result to forrest(). Column names estimate, conf.low, and conf.high map directly to the estimate, lower, and upper arguments.

fit_lm <- lm(sbp ~ age + female + bmi + smoker + activity, data = cohort)

coefs <- broom::tidy(fit_lm, conf.int = TRUE)
coefs <- coefs[coefs$term != "(Intercept)", ]

# Tidy labels
coefs$term <- c(
  "Age (per 1 year)",
  "Female sex",
  "BMI (per 1 kg/m²)",
  "Current smoker",
  "Physically active"
)

# Formatted coefficient + CI text for the right-hand column
coefs$coef_ci <- sprintf(
  "%.2f (%.2f, %.2f)",
  coefs$estimate, coefs$conf.low, coefs$conf.high
)

forrest(
  coefs,
  estimate = "estimate",
  lower    = "conf.low",
  upper    = "conf.high",
  label    = "term",
  header   = "Predictor",
  cols     = c("Coef (95% CI)" = "coef_ci"),
  widths   = c(3, 4, 2.5),
  xlab     = "Regression coefficient (95% CI)",
  stripe   = TRUE
)

For logistic regression, set exponentiate = TRUE in broom::tidy() to get odds ratios directly, then set log_scale = TRUE and ref_line = 1 in forrest():

fit_glm <- glm(htn ~ age + female + bmi + smoker + activity,
               data   = cohort,
               family = binomial)

or_coefs <- broom::tidy(fit_glm, conf.int = TRUE, exponentiate = TRUE)
or_coefs <- or_coefs[or_coefs$term != "(Intercept)", ]
or_coefs$term <- c(
  "Age (per 1 year)",
  "Female sex",
  "BMI (per 1 kg/m²)",
  "Current smoker",
  "Physically active"
)

# Add formatted OR column
or_coefs$or_text <- sprintf(
  "%.2f (%.2f, %.2f)",
  or_coefs$estimate,
  or_coefs$conf.low,
  or_coefs$conf.high
)

forrest(
  or_coefs,
  estimate   = "estimate",
  lower      = "conf.low",
  upper      = "conf.high",
  label      = "term",
  log_scale  = TRUE,
  ref_line   = 1,
  header     = "Predictor",
  cols       = c("OR (95% CI)" = "or_text"),
  widths     = c(3, 4, 2),
  xlab       = "Odds ratio (95% CI)"
)

Same predictor across adjustment models

A common reporting pattern is to show how an association changes as covariates are progressively added. Extract the row for the exposure of interest from each model and combine.

m1 <- glm(htn ~ age,                          data = cohort, family = binomial)
m2 <- glm(htn ~ age + female,                 data = cohort, family = binomial)
m3 <- glm(htn ~ age + female + bmi + smoker + activity, data = cohort, family = binomial)

# Extract the age row from each model
extract_age <- function(mod, label) {
  row <- broom::tidy(mod, conf.int = TRUE, exponentiate = TRUE)
  row <- row[row$term == "age", ]
  row$model <- label
  row
}

age_across_models <- rbind(
  extract_age(m1, "Model 1: unadjusted"),
  extract_age(m2, "Model 2: + sex"),
  extract_age(m3, "Model 3: + sex, BMI, smoking, activity")
)

forrest(
  age_across_models,
  estimate   = "estimate",
  lower      = "conf.low",
  upper      = "conf.high",
  label      = "model",
  log_scale  = TRUE,
  ref_line   = 1,
  xlab       = "Odds ratio for age (per 1 year, 95% CI)"
)

Same predictor across multiple outcomes

To compare one exposure against several outcomes, loop over outcomes, extract the exposure row from each model, and stack into a single data frame. Use group to colour by outcome domain.

outcomes <- list(
  list(var = "sbp",    label = "Systolic BP (mmHg)",       domain = "Continuous"),
  list(var = "htn",    label = "Hypertension (binary)",    domain = "Binary"),
  list(var = "bmi",    label = "BMI (kg/m²)",              domain = "Continuous"),
  list(var = "smoker", label = "Current smoker (binary)",  domain = "Binary")
)

rows <- lapply(outcomes, function(o) {
  if (o$domain == "Continuous") {
    fit <- lm(
      as.formula(paste(o$var, "~ activity + age + female")),
      data = cohort
    )
    r <- broom::tidy(fit, conf.int = TRUE)
  } else {
    fit <- glm(
      as.formula(paste(o$var, "~ activity + age + female")),
      data   = cohort,
      family = binomial
    )
    r <- broom::tidy(fit, conf.int = TRUE, exponentiate = TRUE)
  }
  r <- r[r$term == "activity", ]
  r$outcome <- o$label
  r$domain  <- o$domain
  r
})
multi_out <- do.call(rbind, rows)

forrest(
  multi_out,
  estimate   = "estimate",
  lower      = "conf.low",
  upper      = "conf.high",
  label      = "outcome",
  group      = "domain",
  ref_line   = 0,
  legend_pos = "topright",
  xlab       = "Effect of physical activity (coefficient or OR, 95% CI)",
  title      = "Association of physical activity with multiple outcomes"
)

Note: Because continuous and binary outcomes are on different scales, a single reference line may not be appropriate for all rows. Consider displaying them in separate panels, or use ref_line = NULL and add contextual annotations.

Dose-response (exposure categories)

For categorical exposures (e.g., quartiles, tertiles), the reference category has estimate = NA since no CI can be drawn for it. Use a text column to show the reference label.

# Assign quartiles of age
cohort$age_q <- cut(cohort$age, breaks = quantile(cohort$age, 0:4 / 4),
                    include.lowest = TRUE,
                    labels = c("Q1", "Q2", "Q3", "Q4"))
cohort$age_q <- relevel(cohort$age_q, ref = "Q1")

fit_q <- glm(htn ~ age_q + female + bmi + smoker + activity,
             data = cohort, family = binomial)
q_coefs <- broom::tidy(fit_q, conf.int = TRUE, exponentiate = TRUE)
q_coefs <- q_coefs[grep("^age_q", q_coefs$term), ]

# Build the full table including the reference row
q_plot <- rbind(
  data.frame(
    term      = "age_qQ1",
    estimate  = NA_real_,
    conf.low  = NA_real_,
    conf.high = NA_real_,
    or_ci     = "1.00 (ref)"
  ),
  data.frame(
    term      = q_coefs$term,
    estimate  = q_coefs$estimate,
    conf.low  = q_coefs$conf.low,
    conf.high = q_coefs$conf.high,
    or_ci     = sprintf("%.2f (%.2f, %.2f)",
                        q_coefs$estimate,
                        q_coefs$conf.low,
                        q_coefs$conf.high)
  )
)
q_plot$label  <- c("Q1 (reference)", "Q2", "Q3", "Q4")
q_plot$is_sum <- FALSE

forrest(
  q_plot,
  estimate   = "estimate",
  lower      = "conf.low",
  upper      = "conf.high",
  label      = "label",
  is_summary = "is_sum",
  log_scale  = TRUE,
  ref_line   = 1,
  header     = "Age quartile",
  cols       = c("OR (95% CI)" = "or_ci"),
  widths     = c(2.5, 4, 2),
  xlab       = "Odds ratio for hypertension (95% CI)"
)

Marginal estimates (g-formula, IPW, etc.)

For causal methods that return marginal effects directly (parametric g-formula, inverse probability weighting, targeted maximum likelihood estimation), the output is typically already a data frame of estimates and confidence intervals. Pass it directly to forrest() — no model extraction step is needed.

# Hypothetical marginal risk differences from a g-formula analysis
gform_dat <- data.frame(
  contrast  = c(
    "Active vs. inactive (overall)",
    "  Age < 50",
    "  Age 50–65",
    "  Age > 65"
  ),
  estimate  = c(-0.068, -0.045, -0.072, -0.089),
  lower     = c(-0.105, -0.091, -0.122, -0.148),
  upper     = c(-0.031,  0.001, -0.022, -0.030),
  is_sum    = c(TRUE, FALSE, FALSE, FALSE)
)

forrest(
  gform_dat,
  estimate   = "estimate",
  lower      = "lower",
  upper      = "upper",
  label      = "contrast",
  is_summary = "is_sum",
  ref_line   = 0,
  xlab       = "Marginal risk difference (95% CI)",
  title      = "Effect of physical activity on hypertension risk"
)

Saving plots

Export any plot to a file with save_forrest(). The device is inferred from the file extension (.pdf, .png, .svg, .tiff).

save_forrest(
  file   = "regression_forest.pdf",
  plot   = function() {
    forrest(
      coefs,
      estimate = "estimate",
      lower    = "conf.low",
      upper    = "conf.high",
      label    = "term",
      stripe   = TRUE,
      xlab     = "Regression coefficient (95% CI)"
    )
  },
  width  = 9,
  height = 5
)
